import requests
import pandas as pd
import time
from concurrent.futures import ThreadPoolExecutor, as_completed
from requests.adapters import HTTPAdapter
from google.colab import files

# =====================================================
# CONFIGURACIÓN
# =====================================================

TOKEN = "20c5066cf5c092b6d28afd9a02032bf0"
URL = "https://api.travelpayouts.com/aviasales/v3/prices_for_dates"

# Parámetros base comunes a todas las llamadas
BASE_PARAMS = {
    "currency": "EUR",
    "market": "es",
    "sorting": "price",
    "limit": 1000,
    "token": TOKEN
}

# Número de llamadas en paralelo. Si aparecen muchos errores 429
# (Too Many Requests) o el proceso se vuelve inestable, baja este número.
HILOS = 50

# Pausa (segundos) al inicio de cada llamada para no saturar la API.
# Con muchos hilos conviene que sea pequeña o incluso 0.
PAUSA = 0.05

# Reintentos ante errores 429 (Too Many Requests): con muchos hilos la API
# puede rechazar llamadas. En vez de perder esos vuelos, esperamos y
# reintentamos hasta MAX_REINTENTOS veces.
MAX_REINTENTOS = 3
ESPERA_REINTENTO = 2.0  # segundos de espera antes de reintentar

# Sesión reutilizable: mantiene viva la conexión entre llamadas (más rápido).
# Ajustamos el tamaño del pool de conexiones para que iguale al número de
# hilos; si no, urllib3 avisa "Connection pool is full" y descarta conexiones.
session = requests.Session()
adaptador = HTTPAdapter(pool_connections=HILOS, pool_maxsize=HILOS)
session.mount("https://", adaptador)
session.mount("http://", adaptador)

# =====================================================
# 1) CARGAR EL CSV DE AEROPUERTOS
# =====================================================
# Se lee desde la misma carpeta donde se ejecuta el script.

nombre_csv = "airports_flightable_categorized.csv"

airports = pd.read_csv(nombre_csv, sep=";", encoding="utf-8-sig")

# Nos quedamos solo con lo que necesitamos: el código IATA y sus categorías
airports = airports[["code", "Zone", "Top151"]].dropna(subset=["code"])
airports["code"] = airports["code"].astype(str).str.upper().str.strip()

# =====================================================
# 2) PEDIR ORIGEN Y GRUPO DE DESTINOS
# =====================================================

ORIGEN = input("\nAeropuerto de ORIGEN (IATA): ").upper().strip()

print("\nElige el grupo de destinos:")
print("  1 - Zona: Europe")
print("  2 - Zona: Asia")
print("  3 - Zona: America")
print("  4 - Zona: Africa / Oceania")
print("  5 - Top 151")

opcion = input("\nOpción (1-5): ").strip()

mapa_zonas = {
    "1": "Europe",
    "2": "Asia",
    "3": "America",
    "4": "Africa / Oceania",
}

if opcion in mapa_zonas:
    zona = mapa_zonas[opcion]
    seleccionados = airports[airports["Zone"] == zona]
    grupo_nombre = zona
elif opcion == "5":
    seleccionados = airports[airports["Top151"] == 1]
    grupo_nombre = "Top151"
else:
    raise ValueError("Opción no válida. Debe ser un número del 1 al 5.")

# Lista de destinos = todos los del grupo menos el origen
destinos = [c for c in seleccionados["code"].unique() if c != ORIGEN]

print(f"\nOrigen: {ORIGEN}")
print(f"Grupo elegido: {grupo_nombre}")
print(f"Destinos a consultar: {len(destinos)}")
print(f"Llamadas totales estimadas: {len(destinos) * 3}")

# =====================================================
# 3) FUNCIÓN PARA HACER UNA LLAMADA
# =====================================================

def consultar(origin, destination, one_way, tipo_llamada):
    """Hace una llamada a la API y devuelve una lista de dicts con los vuelos,
    añadiendo columnas de contexto para identificar la ruta y el tipo.
    Reintenta automáticamente si la API responde 429 (Too Many Requests)."""
    params = dict(BASE_PARAMS)
    params["origin"] = origin
    params["destination"] = destination
    params["one_way"] = "true" if one_way else "false"

    data = None
    for intento in range(1, MAX_REINTENTOS + 1):
        if PAUSA:
            time.sleep(PAUSA)
        try:
            resp = session.get(URL, params=params, timeout=30)

            # Si nos limitan (429), esperamos y reintentamos
            if resp.status_code == 429:
                if intento < MAX_REINTENTOS:
                    time.sleep(ESPERA_REINTENTO)
                    continue
                else:
                    print(f"  429 (límite) en {origin}->{destination} ({tipo_llamada}) tras {MAX_REINTENTOS} intentos")
                    return []

            data = resp.json()
            break
        except Exception as e:
            if intento < MAX_REINTENTOS:
                time.sleep(ESPERA_REINTENTO)
                continue
            print(f"  ERROR en {origin}->{destination} ({tipo_llamada}): {e}")
            return []

    filas = []
    if isinstance(data, dict) and data.get("data"):
        for vuelo in data["data"]:
            registro = dict(vuelo)
            registro["ruta_origen"] = origin
            registro["ruta_destino"] = destination
            registro["one_way"] = params["one_way"]
            registro["tipo_llamada"] = tipo_llamada
            filas.append(registro)
    return filas

# =====================================================
# 4) RECORRER TODOS LOS DESTINOS (3 LLAMADAS CADA UNO) EN PARALELO
# =====================================================

# Preparamos la lista de todas las llamadas a hacer (3 por destino).
tareas = []
for DESTINO in destinos:
    # Llamada 1: ida (origen -> destino, solo ida)
    tareas.append((ORIGEN, DESTINO, True, "OW_IDA"))
    # Llamada 2: vuelta (destino -> origen, solo ida)
    tareas.append((DESTINO, ORIGEN, True, "OW_VUELTA"))
    # Llamada 3: redondo (origen -> destino, ida y vuelta)
    tareas.append((ORIGEN, DESTINO, False, "RD"))

resultados = []
total_tareas = len(tareas)
completadas = 0

# Lanzamos las llamadas en paralelo con un pool de hilos.
with ThreadPoolExecutor(max_workers=HILOS) as executor:
    futuros = [
        executor.submit(consultar, origin, destination, one_way, tipo)
        for (origin, destination, one_way, tipo) in tareas
    ]

    for futuro in as_completed(futuros):
        resultados += futuro.result()
        completadas += 1
        if completadas % 25 == 0 or completadas == total_tareas:
            print(f"  {completadas}/{total_tareas} llamadas completadas")

print(f"\n\nTotal de vuelos recopilados: {len(resultados)}")

# =====================================================
# 5) GUARDAR Y DESCARGAR EL CSV
# =====================================================

if resultados:
    df = pd.DataFrame(resultados)

    # Ponemos primero las columnas de contexto para que sea más legible
    cols_contexto = ["ruta_origen", "ruta_destino", "one_way", "tipo_llamada"]
    otras = [c for c in df.columns if c not in cols_contexto]
    df = df[cols_contexto + otras]

    print(f"\nSe encontraron {len(df)} vuelos en total.")
    display(df)

    nombre_archivo = f"resultado_{ORIGEN}_{grupo_nombre.replace(' / ', '_').replace(' ', '')}.csv"

    df.to_csv(
        nombre_archivo,
        sep=";",
        index=False,
        encoding="utf-8-sig"
    )

    files.download(nombre_archivo)
    print(f"\nArchivo descargado: {nombre_archivo}")
else:
    print("\nSin resultados para ninguna ruta.")
